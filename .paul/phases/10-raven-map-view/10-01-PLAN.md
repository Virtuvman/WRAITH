    def _load_raven_data(mock: bool = True) -> pd.DataFrame:
      If mock:
        combined = fetch_all_feeds(mock=True)  # 55 records across all sources
        shodan_mock = generate_mock_raven(n=10)  # additional Shodan mock pins
        df = pd.concat([combined, shodan_mock], ignore_index=True)
      Else:
        frames = []
        # Scan data/ for Shodan/FOFA JSON files
        for p in sorted(Path("data").glob("*.json")):
          if p.name == "youtube_cameras.json":
            continue  # seed file, not a RAVEN export
          frames.append(load_raven_file(str(p)))
        # Pull live feeds (may fail silently)
        frames.append(fetch_all_feeds(mock=False))
        if not frames or all(f.empty for f in frames):
          return pd.DataFrame(columns=RAVEN_SCHEMA_COLUMNS)
        df = pd.concat([f for f in frames if not f.empty], ignore_index=True)
      # Run classifier — adds camera_type column, doesn't filter
      df = tag_camera_type(df)
      return df

    --- render_raven_view ---
    def render_raven_view():
      st.markdown('<div class="section-label">RAVEN — Camera Intelligence</div>',
                  unsafe_allow_html=True)

      # ── Load controls ─────────────────────────────────────────────────────
      col_mock, col_load = st.columns([3, 1])
      with col_mock:
        mock = st.toggle("Mock data (no network / API calls)",
                         value=st.session_state.raven_mock,
                         key="raven_mock_toggle")
        st.session_state.raven_mock = mock
      with col_load:
        if st.button("Load RAVEN Data", use_container_width=True, key="raven_load_btn"):
          with st.spinner("Loading RAVEN feeds…"):
            st.session_state.raven_df = _load_raven_data(mock=mock)

      raven_df = st.session_state.raven_df

      # ── Empty state ────────────────────────────────────────────────────────
      if raven_df is None or raven_df.empty:
        st.info(
          "No RAVEN data loaded. Click **Load RAVEN Data** to pull "
          "municipal feeds + YouTube thumbnails + any Shodan/CSV files in data/."
        )
        return

      # ── KPI row ────────────────────────────────────────────────────────────
      total_pins = len(raven_df)
      with_img = int(raven_df["image_b64"].notna().sum())
      sources_list = sorted(raven_df["source"].dropna().unique())
      countries = set()
      for meta in raven_df["metadata"].dropna():
        if isinstance(meta, dict) and meta.get("country"):
          countries.add(meta["country"])

      k1, k2, k3, k4 = st.columns(4)
      k1.metric("Camera Pins", total_pins)
      k2.metric("With Screenshot", with_img)
      k3.metric("Sources", len(sources_list))
      k4.metric("Countries", len(countries) or "—")

      # ── Source legend ──────────────────────────────────────────────────────
      SOURCE_COLORS = {
        "shodan": "#ef4444",
        "fofa": "#f97316",
        "municipal": "#3b82f6",
        "youtube": "#a855f7",
        "csv": "#22c55e",
      }
      DEFAULT_COLOR = "#94a3b8"

      legend_parts = " &nbsp; ".join(
        f'<span style="color:{SOURCE_COLORS.get(s, DEFAULT_COLOR)}">●</span> {s}'
        for s in sources_list
      )
      st.markdown(f'<div style="font-size:0.8rem;opacity:0.75">{legend_parts}</div>',
                  unsafe_allow_html=True)

      # ── Folium map ─────────────────────────────────────────────────────────
      if not FOLIUM_AVAILABLE:
        st.warning("folium not installed — cannot render RAVEN map.")
        return

      valid = raven_df.dropna(subset=["lat", "lon"])
      if valid.empty:
        st.warning("No records with valid lat/lon coordinates.")
        return

      center_lat = float(valid["lat"].median())
      center_lon = float(valid["lon"].median())

      m = folium.Map(
        location=[center_lat, center_lon],
        zoom_start=4,
        tiles="CartoDB dark_matter",
        attr="CartoDB",
      )

      for _, row in valid.iterrows():
        src = str(row.get("source") or "")
        color = SOURCE_COLORS.get(src, DEFAULT_COLOR)
        row_dict = row.to_dict()
        popup_html = build_popup_html(row_dict)

        folium.CircleMarker(
          location=[float(row["lat"]), float(row["lon"])],
          radius=7,
          color=color,
          fill=True,
          fill_color=color,
          fill_opacity=0.8,
          weight=2,
          popup=folium.Popup(popup_html, max_width=300),
          tooltip=folium.Tooltip(
            f"{row.get('label','') or row.get('ip','') or 'camera'} · {src}",
            sticky=False,
          ),
        ).add_to(m)

      st_folium(m, use_container_width=True, height=560, returned_objects=[])
