    Create the public feed integration module. No streamlit imports.
    Imports: requests, json, base64, logging, hashlib, random, pathlib, pandas

    --- CONSTANTS ---
    CALTRANS_DISTRICTS = [3, 4, 7]  # Sacramento, Bay Area, Los Angeles
    CALTRANS_URL = "https://cwwp2.dot.ca.gov/data/d{d}/cctv/cctvStatusD0{d}.json"
    WSDOT_URL = "https://www.wsdot.wa.gov/traffic/api/HighwayCameras/HighwayCamerasREST.svc/GetCamerasAsJSON"
    YOUTUBE_CAMERAS_FILE = "data/youtube_cameras.json"
    YOUTUBE_THUMB_URL = "https://img.youtube.com/vi/{video_id}/hqdefault.jpg"

    --- HELPER ---
    _empty_raven() -> dict:
      Return {col: None for col in RAVEN_SCHEMA_COLUMNS} with tags=[], metadata={}

    _mock_seed(label: str) -> random.Random:
      Return Random seeded from hash of label string — deterministic per source.

    --- CALTRANS ---
    _parse_caltrans_record(row: dict) -> dict:
      Parse one Caltrans CCTV JSON record. Structure varies by district:
        cctv = row.get("cctv", row)
        location = cctv.get("location", {})
        image_data = cctv.get("imageData", {})
        static_img = image_data.get("static", {})

        lat = _safe_float(location.get("locationLat"))
        lon = _safe_float(location.get("locationLong"))
        name = location.get("locationName") or location.get("linearReference", "")
        img_url = static_img.get("currentImageURL") or image_data.get("streamingVideoURL")

        out = _empty_raven()
        out["source"] = "municipal"
        out["lat"] = lat
        out["lon"] = lon
        out["label"] = f"Caltrans: {name}"
        out["image_url"] = img_url
        out["tags"] = ["traffic", "caltrans", "municipal"]
        out["metadata"] = {"provider": "Caltrans", "district": ""}
        return out

    fetch_caltrans_cameras(district: int = 4, timeout: int = 10, mock: bool = False) -> pd.DataFrame:
      If mock:
        rng = _mock_seed(f"caltrans_{district}")
        Generate 8 mock records around Bay Area (37.8, -122.2) with ±0.1 scatter.
        Each has source="municipal", image_url=f"https://mock.caltrans.gov/d{district}/{i}.jpg",
        label=f"Caltrans D{district} Camera {i}", tags=["traffic","caltrans","municipal"]
        Return DataFrame with RAVEN_SCHEMA_COLUMNS.
      Else:
        url = CALTRANS_URL.format(d=district, d=f"{district:02d}")  — note: format both {d} references
        Actually: url = f"https://cwwp2.dot.ca.gov/data/d{district}/cctv/cctvStatusD{district:02d}.json"
        GET with timeout, parse JSON.
        Navigate to records: data = resp.json(); rows = data.get("data", {}).get("row", [])
        Parse each with _parse_caltrans_record.
        Return DataFrame. On any exception log + return empty DataFrame.

    --- WSDOT ---
    _parse_wsdot_record(cam: dict) -> dict:
      out = _empty_raven()
      out["source"] = "municipal"
      out["ip"] = None
      out["lat"] = _safe_float(cam.get("Latitude"))
      out["lon"] = _safe_float(cam.get("Longitude"))
      out["label"] = f"WSDOT: {cam.get('Title', '')}"
      out["image_url"] = cam.get("ImageURL") or cam.get("ImageURLSecure")
      out["tags"] = ["traffic", "wsdot", "municipal"]
      out["timestamp"] = cam.get("LastUpdated")
      out["metadata"] = {"provider": "WSDOT", "road": cam.get("RoadName", "")}
      return out

    fetch_wsdot_cameras(timeout: int = 10, mock: bool = False) -> pd.DataFrame:
      If mock:
        rng = _mock_seed("wsdot")
        Generate 8 mock records around Seattle (47.6, -122.3) with ±0.1 scatter.
        Return DataFrame with RAVEN_SCHEMA_COLUMNS.
      Else:
        GET WSDOT_URL with timeout.
        Response is a JSON array of camera objects.
        Parse each with _parse_wsdot_record.
        Return DataFrame. On exception log + return empty DataFrame.

    --- YOUTUBE ---
    _load_youtube_cameras(cameras_file: str = YOUTUBE_CAMERAS_FILE) -> list:
      Load JSON file, return data["cameras"] list. Return [] on error.

    _fetch_youtube_thumb_b64(video_id: str, timeout: int = 8) -> str | None:
      GET YOUTUBE_THUMB_URL.format(video_id=video_id) with timeout.
      On 200: base64.b64encode(resp.content).decode("utf-8")
      On any error: return None — never raise.

    fetch_youtube_thumbnails(
        cameras_file: str = YOUTUBE_CAMERAS_FILE,
        timeout: int = 10,
        mock: bool = False
    ) -> pd.DataFrame:
      cameras = _load_youtube_cameras(cameras_file)
      If not cameras: return empty DataFrame with RAVEN_SCHEMA_COLUMNS.

      rows = []
      For each cam in cameras:
        out = _empty_raven()
        out["source"] = "youtube"
        out["lat"] = _safe_float(cam.get("lat"))
        out["lon"] = _safe_float(cam.get("lon"))
        out["label"] = cam.get("name") or cam.get("city") or "YouTube camera"
        out["tags"] = cam.get("tags", ["traffic", "youtube"])
        out["metadata"] = {"city": cam.get("city"), "country": cam.get("country"), "video_id": cam.get("video_id")}

        If mock:
          out["image_b64"] = MOCK_B64_PNG
          out["image_url"] = None
        Else:
          vid = cam.get("video_id", "")
          if vid and not vid.startswith("REPLACE"):
            out["image_b64"] = _fetch_youtube_thumb_b64(vid, timeout)
          out["image_url"] = f"https://img.youtube.com/vi/{vid}/hqdefault.jpg" if vid else None

        rows.append(out)

      Return pd.DataFrame(rows, columns=RAVEN_SCHEMA_COLUMNS)

    --- COMBINED ---
    fetch_all_feeds(
        sources: list | None = None,
        caltrans_districts: list = CALTRANS_DISTRICTS,
        youtube_file: str = YOUTUBE_CAMERAS_FILE,
        timeout: int = 10,
        mock: bool = False
    ) -> pd.DataFrame:
      sources defaults to ["caltrans", "wsdot", "youtube"] if None.
      Collect DataFrames from enabled sources.
      Concatenate with pd.concat, reset_index drop=True.
      Return empty DataFrame with RAVEN_SCHEMA_COLUMNS if all sources empty.
      Never raise.

    --- _safe_float ---
    Same as in raven_ingest.py — copy locally:
    def _safe_float(val):
        if val is None: return None
        try: return float(val)
        except: return None

    def load_raven_csv(src) -> pd.DataFrame:
      """Load a CSV (filepath string, Path, or file-like) and normalize to RAVEN schema.

      Applies _CSV_COLUMN_ALIASES for flexible column name variants.
      Any RAVEN_SCHEMA_COLUMNS not present are added as None.
      """
      try:
          df = pd.read_csv(src)
          # Normalize column names: lowercase + strip
          df.columns = [c.strip().lower() for c in df.columns]
          # Apply alias map
          df = df.rename(columns={k: v for k, v in _CSV_COLUMN_ALIASES.items() if k in df.columns})
          # Add missing schema columns
          for col in RAVEN_SCHEMA_COLUMNS:
              if col not in df.columns:
                  df[col] = None
          # Coerce lat/lon
          df["lat"] = df["lat"].apply(_safe_float)
          df["lon"] = df["lon"].apply(_safe_float)
          # Default tags to empty list where None
          df["tags"] = df["tags"].apply(lambda v: v if isinstance(v, list) else [])
          # Default metadata to empty dict where None
          df["metadata"] = df["metadata"].apply(lambda v: v if isinstance(v, dict) else {})
          return df[RAVEN_SCHEMA_COLUMNS].reset_index(drop=True)
      except Exception as exc:
          log.error("raven_ingest.load_raven_csv error: %s", exc)
          return pd.DataFrame(columns=RAVEN_SCHEMA_COLUMNS)
